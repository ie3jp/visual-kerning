For Engineers

Zero dependencies, framework-agnostic. ESM only.

npm install visual-kerning
yarn add visual-kerning
pnpm add visual-kerning
<script type="module">
  import { visualKerning } from
    'https://esm.sh/visual-kerning@0.11.0'
</script>

Usage

Edit in dev / staging

import { visualKerning } from 'visual-kerning'

const editor = visualKerning({ editable: true })
editor.mount()

Apply in production

const editor = visualKerning({
  editable: false,
  kerning: kerningData
})
editor.mount()

Edit in dev / staging

<script type="module">
import { visualKerning } from
  'https://esm.sh/visual-kerning@0.11.0'

const editor = visualKerning({ editable: true })
editor.mount()
</script>

Apply in production

<script type="module">
import { visualKerning } from
  'https://esm.sh/visual-kerning@0.11.0'

const kerningData = await
  fetch('/kerning-export.json')
    .then(r => r.json())

const editor = visualKerning({
  editable: false,
  kerning: kerningData
})
editor.mount()
</script>
